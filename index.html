class BecomeFriends
  def initialize(user_id, other_user_id)
    @user       = User.find(user_id)
    @other_user = User.find(other_user_id)

    @user.extend(FriendMakerRole)
  end

  def execute
    @user.become_friends_with(@other_user)
  end

  def self.build_new_for(user)
    Friendship.new(user_id: user)
  end

  module FriendMakerRole
    def become_friends_with(other_user)
      Friendship.create(user_1: other_user, user_2: self)
    end
  end
end

class FriendshipsController < ApplicationController
  def edit
    @friendship = BecomeFriends.build_new_for(current_user)
  end

  def create
    @friendship = BecomeFriends.new(current_user.id, params[:user_id]).execute

    if @friendship
      redirect_to action: :index, notice: 'Made friends!'
    else
      render :edit
    end
  end
end
